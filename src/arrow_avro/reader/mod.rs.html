18//! Avro reader
19//!
20//! Facilities to read Apache Avro–encoded data into Arrow's `RecordBatch` format.
21//!
22//! This module exposes three layers of the API surface, from highest to lowest-level:
23//!
24//! * `ReaderBuilder`: configures how Avro is read (batch size, strict union handling,
25//!   string representation, reader schema, etc.) and produces either:
26//!   * a `Reader` for **Avro Object Container Files (OCF)** read from any `BufRead`, or
27//!   * a low-level `Decoder` for **single‑object encoded** Avro bytes and Confluent
28//!     **Schema Registry** framed messages.
29//! * `Reader`: a convenient, synchronous iterator over `RecordBatch` decoded from an OCF
30//!   input. Implements [`Iterator<Item = Result<RecordBatch, ArrowError>>`] and
31//!   `RecordBatchReader`.
32//! * `Decoder`: a push‑based row decoder that consumes raw Avro bytes and yields ready
33//!   `RecordBatch` values when batches fill. This is suitable for integrating with async
34//!   byte streams, network protocols, or other custom data sources.
35//!
36//! ## Encodings and when to use which type
37//!
38//! * **Object Container File (OCF)**: A self‑describing file format with a header containing
39//!   the writer schema, optional compression codec, and a sync marker, followed by one or
40//!   more data blocks. Use `Reader` for this format. See the Avro specification for the
41//!   structure of OCF headers and blocks. <https://avro.apache.org/docs/1.11.1/specification/>
42//! * **Single‑Object Encoding**: A stream‑friendly framing that prefixes each record body with
43//!   the 2‑byte magic `0xC3 0x01` followed by a schema fingerprint. Use `Decoder` with a
44//!   populated `SchemaStore` to resolve fingerprints to full
45//!   schemas. <https://avro.apache.org/docs/1.11.1/specification/>
46//! * **Confluent Schema Registry wire format**: A 1‑byte magic `0x00`, a 4‑byte big‑endian
47//!   schema ID, then the Avro‑encoded body. Use `Decoder` with a
48//!   `SchemaStore` configured for `FingerprintAlgorithm::None`
49//!   and entries keyed by `Fingerprint::Id`. Confluent docs
50//!   describe this framing.
51//!
52//! ## Basic file usage (OCF)
53//!
54//! Use `ReaderBuilder::build` to construct a `Reader` from any `BufRead`, such as a
55//! `BufReader<File>`. The reader yields `RecordBatch` values you can iterate over or collect.
56//!
57//! ```no_run
58//! use std::fs::File;
59//! use std::io::BufReader;
60//! use arrow_array::RecordBatch;
61//! use arrow_avro::reader::ReaderBuilder;
62//!
63//! // Locate a test file (mirrors Arrow's test data layout)
64//! let path = "avro/alltypes_plain.avro";
65//! let path = std::env::var("ARROW_TEST_DATA")
66//!     .map(|dir| format!("{dir}/{path}"))
67//!     .unwrap_or_else(|_| format!("../testing/data/{path}"));
68//!
69//! let file = File::open(path).unwrap();
70//! let mut reader = ReaderBuilder::new().build(BufReader::new(file)).unwrap();
71//!
72//! // Iterate batches
73//! let mut num_rows = 0usize;
74//! while let Some(batch) = reader.next() {
75//!     let batch: RecordBatch = batch.unwrap();
76//!     num_rows += batch.num_rows();
77//! }
78//! println!("decoded {num_rows} rows");
79//! ```
80//!
81//! ## Streaming usage (single‑object / Confluent)
82//!
83//! The `Decoder` lets you integrate Avro decoding with **any** source of bytes by
84//! periodically calling `Decoder::decode` with new data and calling `Decoder::flush`
85//! to get a `RecordBatch` once at least one row is complete.
86//!
87//! The example below shows how to decode from an arbitrary stream of `bytes::Bytes` using
88//! `futures` utilities. Note: this is illustrative and keeps a single in‑memory `Bytes`
89//! buffer for simplicity—real applications typically maintain a rolling buffer.
90//!
91//! ```no_run
92//! use bytes::{Buf, Bytes};
93//! use futures::{Stream, StreamExt};
94//! use std::task::{Poll, ready};
95//! use arrow_array::RecordBatch;
96//! use arrow_schema::ArrowError;
97//! use arrow_avro::reader::Decoder;
98//!
99//! /// Decode a stream of Avro-framed bytes into RecordBatch values.
100//! fn decode_stream<S: Stream<Item = Bytes> + Unpin>(
101//!     mut decoder: Decoder,
102//!     mut input: S,
103//! ) -> impl Stream<Item = Result<RecordBatch, ArrowError>> {
104//!     let mut buffered = Bytes::new();
105//!     futures::stream::poll_fn(move |cx| {
106//!         loop {
107//!             if buffered.is_empty() {
108//!                 buffered = match ready!(input.poll_next_unpin(cx)) {
109//!                     Some(b) => b,
110//!                     None => break, // EOF
111//!                 };
112//!             }
113//!             // Feed as much as possible
114//!             let decoded = match decoder.decode(buffered.as_ref()) {
115//!                 Ok(n) => n,
116//!                 Err(e) => return Poll::Ready(Some(Err(e))),
117//!             };
118//!             let read = buffered.len();
119//!             buffered.advance(decoded);
120//!             if decoded != read {
121//!                 // decoder made partial progress; request more bytes
122//!                 break
123//!             }
124//!         }
125//!         // Return a batch if one or more rows are complete
126//!         Poll::Ready(decoder.flush().transpose())
127//!     })
128//! }
129//! ```
130//!
131//! ### Building a `Decoder` for **single‑object encoding** (Rabin fingerprints)
132//!
133//! ```no_run
134//! use arrow_avro::schema::{AvroSchema, SchemaStore};
135//! use arrow_avro::reader::ReaderBuilder;
136//!
137//! // Build a SchemaStore and register known writer schemas
138//! let mut store = SchemaStore::new(); // Rabin by default
139//! let user_schema = AvroSchema::new(r#"{"type":"record","name":"User","fields":[
140//!     {"name":"id","type":"long"},{"name":"name","type":"string"}]}"#.to_string());
141//! let _fp = store.register(user_schema).unwrap(); // computes Rabin CRC-64-AVRO
142//!
143//! // Build a Decoder that expects single-object encoding (0xC3 0x01 + fingerprint and body)
144//! let decoder = ReaderBuilder::new()
145//!     .with_writer_schema_store(store)
146//!     .with_batch_size(1024)
147//!     .build_decoder()
148//!     .unwrap();
149//! // Feed decoder with framed bytes (not shown; see `decode_stream` above).
150//! ```
151//!
152//! ### Building a `Decoder` for **Confluent Schema Registry** framed messages
153//!
154//! ```no_run
155//! use arrow_avro::schema::{AvroSchema, SchemaStore, Fingerprint, FingerprintAlgorithm};
156//! use arrow_avro::reader::ReaderBuilder;
157//!
158//! // Confluent wire format uses a magic 0x00 byte + 4-byte schema id (big-endian).
159//! // Create a store keyed by `Fingerprint::Id` and pre-populate with known schemas.
160//! let mut store = SchemaStore::new_with_type(FingerprintAlgorithm::None);
161//!
162//! // Suppose registry ID 42 corresponds to this Avro schema:
163//! let avro = AvroSchema::new(r#"{"type":"string"}"#.to_string());
164//! store.set(Fingerprint::Id(42), avro).unwrap();
165//!
166//! // Build a Decoder that understands Confluent framing
167//! let decoder = ReaderBuilder::new()
168//!     .with_writer_schema_store(store)
169//!     .build_decoder()
170//!     .unwrap();
171//! // Feed decoder with 0x00 + [id:4] + Avro body frames.
172//! ```
173//!
174//! ## Schema evolution and batch boundaries
175//!
176//! `Decoder` supports mid‑stream schema changes when the input framing carries a schema
177//! fingerprint (single‑object or Confluent). When a new fingerprint is observed:
178//!
179//! * If the current `RecordBatch` is **empty**, the decoder switches to the new schema
180//!   immediately.
181//! * If not, the decoder finishes the current batch first and only then switches.
182//!
183//! Consequently, the schema of batches produced by `Decoder::flush` may change over time,
184//! and `Decoder` intentionally does **not** implement `RecordBatchReader`. In contrast,
185//! `Reader` (OCF) has a single writer schema for the entire file and therefore implements
186//! `RecordBatchReader`.
187//!
188//! ## Performance & memory
189//!
190//! * `batch_size` controls the maximum number of rows per `RecordBatch`. Larger batches
191//!   amortize per‑batch overhead; smaller batches reduce peak memory usage and latency.
192//! * When `utf8_view` is enabled, string columns use Arrow’s `StringViewArray`, which can
193//!   reduce allocations for short strings.
194//! * For OCF, blocks may be compressed `Reader` will decompress using the codec specified
195//!   in the file header and feed uncompressed bytes to the row `Decoder`.
196//!
197//! ## Error handling
198//!
199//! * Incomplete inputs return parse errors with "Unexpected EOF"; callers typically provide
200//!   more bytes and try again.
201//! * If a fingerprint is unknown to the provided `SchemaStore`, decoding fails with a
202//!   descriptive error. Populate the store up front to avoid this.
203//!
204//! ---

225/// Read the Avro file header (magic, metadata, sync marker) from `reader`.
226fn read_header<R: BufRead>(mut reader: R) -> Result<Header, ArrowError> {
227    let mut decoder = HeaderDecoder::default();
228    loop {
229        let buf = reader.fill_buf()?;
230        if buf.is_empty() {
231            break;
232        }
233        let read = buf.len();
234        let decoded = decoder.decode(buf)?;
235        reader.consume(decoded);
236        if decoded != read {
237            break;
238        }
239    }
240    decoder.flush().ok_or_else(|| {
241        ArrowError::ParseError("Unexpected EOF while reading Avro header".to_string())
242    })
243}

254/// A low‑level, push‑based decoder from Avro bytes to Arrow `RecordBatch`.
255///
256/// `Decoder` is designed for **streaming** scenarios:
257///
258/// * You *feed* freshly received bytes using `Self::decode`, potentially multiple times,
259///   until at least one row is complete.
260/// * You then *drain* completed rows with `Self::flush`, which yields a `RecordBatch`
261///   if any rows were finished since the last flush.
262///
263/// Unlike `Reader`, which is specialized for Avro **Object Container Files**, `Decoder`
264/// understands **framed single‑object** inputs and **Confluent Schema Registry** messages,
265/// switching schemas mid‑stream when the framing indicates a new fingerprint.
266///
267/// ### Supported prefixes
268///
269/// On each new row boundary, `Decoder` tries to match one of the following "prefixes":
270///
271/// * **Single‑Object encoding**: magic `0xC3 0x01` + schema fingerprint (length depends on
272///   the configured `FingerprintAlgorithm`); see `SINGLE_OBJECT_MAGIC`.
273/// * **Confluent wire format**: magic `0x00` + 4‑byte big‑endian schema id; see
274///   `CONFLUENT_MAGIC`.
275///
276/// The active fingerprint determines which cached row decoder is used to decode the following
277/// record body bytes.
278///
279/// ### Schema switching semantics
280///
281/// When a new fingerprint is observed:
282///
283/// * If the current batch is empty, the decoder switches immediately;
284/// * Otherwise, the current batch is finalized on the next `flush` and only then
285///   does the decoder switch to the new schema. This guarantees that a single `RecordBatch`
286///   never mixes rows with different schemas.
287///
288/// ### Examples
289///
290/// Build a `Decoder` for single‑object encoding using a `SchemaStore` with Rabin fingerprints:
291///
292/// ```no_run
293/// use arrow_avro::schema::{AvroSchema, SchemaStore};
294/// use arrow_avro::reader::ReaderBuilder;
295///
296/// let mut store = SchemaStore::new(); // Rabin by default
297/// let avro = AvroSchema::new(r#""string""#.to_string());
298/// let _fp = store.register(avro).unwrap();
299///
300/// let mut decoder = ReaderBuilder::new()
301///     .with_writer_schema_store(store)
302///     .with_batch_size(512)
303///     .build_decoder()
304///     .unwrap();
305///
306/// // Feed bytes (framed as 0xC3 0x01 + fingerprint and body)
307/// // decoder.decode(&bytes)?;
308/// // if let Some(batch) = decoder.flush()? { /* process */ }
309/// ```
310///
311/// Build a `Decoder` for Confluent Registry messages (magic 0x00 + 4‑byte id):
312///
313/// ```no_run
314/// use arrow_avro::schema::{AvroSchema, SchemaStore, Fingerprint, FingerprintAlgorithm};
315/// use arrow_avro::reader::ReaderBuilder;
316///
317/// let mut store = SchemaStore::new_with_type(FingerprintAlgorithm::None);
318/// store.set(Fingerprint::Id(7), AvroSchema::new(r#""long""#.to_string())).unwrap();
319///
320/// let mut decoder = ReaderBuilder::new()
321///     .with_writer_schema_store(store)
322///     .build_decoder()
323///     .unwrap();
324/// ```
325#[derive(Debug)]
326pub struct Decoder {
327    active_decoder: RecordDecoder,
328    active_fingerprint: Option<Fingerprint>,
329    batch_size: usize,
330    remaining_capacity: usize,
331    cache: IndexMap<Fingerprint, RecordDecoder>,
332    fingerprint_algorithm: FingerprintAlgorithm,
333    utf8_view: bool,
334    strict_mode: bool,
335    pending_schema: Option<(Fingerprint, RecordDecoder)>,
336    awaiting_body: bool,
337}
338
339impl Decoder {
340    /// Returns the Arrow schema for the rows decoded by this decoder.
341    ///
342    /// **Note:** With single‑object or Confluent framing, the schema may change
343    /// at a row boundary when the input indicates a new fingerprint.
344    pub fn schema(&self) -> SchemaRef {
345        self.active_decoder.schema().clone()
346    }
347
348    /// Returns the configured maximum number of rows per batch.
349    pub fn batch_size(&self) -> usize {
350        self.batch_size
351    }
352
353    /// Feed a chunk of bytes into the decoder.
354    ///
355    /// This will:
356    ///
357    /// * Decode at most `Self::batch_size` rows;
358    /// * Return the number of input bytes **consumed** from `data` (which may be 0 if more
359    ///   bytes are required, or less than `data.len()` if a prefix/body straddles the
360    ///   chunk boundary);
361    /// * Defer producing a `RecordBatch` until you call `Self::flush`.
362    ///
363    /// # Returns
364    /// The number of bytes consumed from `data`.
365    ///
366    /// # Errors
367    /// Returns an error if:
368    ///
369    /// * The input indicates an unknown fingerprint (not present in the provided
370    ///   `SchemaStore`;
371    /// * The Avro body is malformed;
372    /// * A strict‑mode union rule is violated (see `ReaderBuilder::with_strict_mode`).
373    pub fn decode(&mut self, data: &[u8]) -> Result<usize, ArrowError> {
374        let mut total_consumed = 0usize;
375        while total_consumed < data.len() && self.remaining_capacity > 0 {
376            if self.awaiting_body {
377                match self.active_decoder.decode(&data[total_consumed..], 1) {
378                    Ok(n) => {
379                        self.remaining_capacity -= 1;
380                        total_consumed += n;
381                        self.awaiting_body = false;
382                        continue;
383                    }
384                    Err(ref e) if is_incomplete_data(e) => break,
385                    err => return err,
386                };
387            }
388            match self.handle_prefix(&data[total_consumed..])? {
389                Some(0) => break, // Insufficient bytes
390                Some(n) => {
391                    total_consumed += n;
392                    self.apply_pending_schema_if_batch_empty();
393                    self.awaiting_body = true;
394                }
395                None => {
396                    return Err(ArrowError::ParseError(
397                        "Missing magic bytes and fingerprint".to_string(),
398                    ))
399                }
400            }
401        }
402        Ok(total_consumed)
403    }
404
405    // Attempt to handle a prefix at the current position.
406    // * Ok(None) – buffer does not start with the prefix.
407    // * Ok(Some(0)) – prefix detected, but the buffer is too short; caller should await more bytes.
408    // * Ok(Some(n)) – consumed `n > 0` bytes of a complete prefix (magic and fingerprint).
409    fn handle_prefix(&mut self, buf: &[u8]) -> Result<Option<usize>, ArrowError> {
410        match self.fingerprint_algorithm {
411            FingerprintAlgorithm::Rabin => {
412                self.handle_prefix_common(buf, &SINGLE_OBJECT_MAGIC, |bytes| {
413                    Fingerprint::Rabin(u64::from_le_bytes(bytes))
414                })
415            }
416            FingerprintAlgorithm::None => {
417                self.handle_prefix_common(buf, &CONFLUENT_MAGIC, |bytes| {
418                    Fingerprint::Id(u32::from_be_bytes(bytes))
419                })
420            }
421            #[cfg(feature = "md5")]
422            FingerprintAlgorithm::MD5 => {
423                self.handle_prefix_common(buf, &SINGLE_OBJECT_MAGIC, |bytes| {
424                    Fingerprint::MD5(bytes)
425                })
426            }
427            #[cfg(feature = "sha256")]
428            FingerprintAlgorithm::SHA256 => {
429                self.handle_prefix_common(buf, &SINGLE_OBJECT_MAGIC, |bytes| {
430                    Fingerprint::SHA256(bytes)
431                })
432            }
433        }
434    }
435
436    /// This method checks for the provided `magic` bytes at the start of `buf` and, if present,
437    /// attempts to read the following fingerprint of `N` bytes, converting it to a
438    /// `Fingerprint` using `fingerprint_from`.
439    fn handle_prefix_common<const MAGIC_LEN: usize, const N: usize>(
440        &mut self,
441        buf: &[u8],
442        magic: &[u8; MAGIC_LEN],
443        fingerprint_from: impl FnOnce([u8; N]) -> Fingerprint,
444    ) -> Result<Option<usize>, ArrowError> {
445        // Need at least the magic bytes to decide
446        // 2 bytes for Avro Spec and 1 byte for Confluent Wire Protocol.
447        if buf.len() < MAGIC_LEN {
448            return Ok(Some(0));
449        }
450        // Bail out early if the magic does not match.
451        if &buf[..MAGIC_LEN] != magic {
452            return Ok(None);
453        }
454        // Try to parse the fingerprint that follows the magic.
455        let consumed_fp = self.handle_fingerprint(&buf[MAGIC_LEN..], fingerprint_from)?;
456        // Convert the inner result into a “bytes consumed” count.
457        // NOTE: Incomplete fingerprint consumes no bytes.
458        Ok(Some(consumed_fp.map_or(0, |n| n + MAGIC_LEN)))
459    }
460
461    // Attempts to read and install a new fingerprint of `N` bytes.
462    //
463    // * Ok(None) – insufficient bytes (`buf.len() < `N`).
464    // * Ok(Some(N)) – fingerprint consumed (always `N`).
465    fn handle_fingerprint<const N: usize>(
466        &mut self,
467        buf: &[u8],
468        fingerprint_from: impl FnOnce([u8; N]) -> Fingerprint,
469    ) -> Result<Option<usize>, ArrowError> {
470        // Need enough bytes to get fingerprint (next N bytes)
471        let Some(fingerprint_bytes) = buf.get(..N) else {
472            return Ok(None); // insufficient bytes
473        };
474        // SAFETY: length checked above.
475        let new_fingerprint = fingerprint_from(fingerprint_bytes.try_into().unwrap());
476        // If the fingerprint indicates a schema change, prepare to switch decoders.
477        if self.active_fingerprint != Some(new_fingerprint) {
478            let Some(new_decoder) = self.cache.shift_remove(&new_fingerprint) else {
479                return Err(ArrowError::ParseError(format!(
480                    "Unknown fingerprint: {new_fingerprint:?}"
481                )));
482            };
483            self.pending_schema = Some((new_fingerprint, new_decoder));
484            // If there are already decoded rows, we must flush them first.
485            // Reducing `remaining_capacity` to 0 ensures `flush` is called next.
486            if self.remaining_capacity < self.batch_size {
487                self.remaining_capacity = 0;
488            }
489        }
490        Ok(Some(N))
491    }
492
493    fn apply_pending_schema(&mut self) {
494        if let Some((new_fingerprint, new_decoder)) = self.pending_schema.take() {
495            if let Some(old_fingerprint) = self.active_fingerprint.replace(new_fingerprint) {
496                let old_decoder = std::mem::replace(&mut self.active_decoder, new_decoder);
497                self.cache.shift_remove(&old_fingerprint);
498                self.cache.insert(old_fingerprint, old_decoder);
499            } else {
500                self.active_decoder = new_decoder;
501            }
502        }
503    }
504
505    fn apply_pending_schema_if_batch_empty(&mut self) {
506        if self.batch_is_empty() {
507            self.apply_pending_schema();
508        }
509    }
510
511    fn flush_and_reset(&mut self) -> Result<Option<RecordBatch>, ArrowError> {
512        if self.batch_is_empty() {
513            return Ok(None);
514        }
515        let batch = self.active_decoder.flush()?;
516        self.remaining_capacity = self.batch_size;
517        Ok(Some(batch))
518    }
519
520    /// Produce a `RecordBatch` if at least one row is fully decoded, returning
521    /// `Ok(None)` if no new rows are available.
522    ///
523    /// If a schema change was detected while decoding rows for the current batch, the
524    /// schema switch is applied **after** flushing this batch, so the **next** batch
525    /// (if any) may have a different schema.
526    pub fn flush(&mut self) -> Result<Option<RecordBatch>, ArrowError> {
527        // We must flush the active decoder before switching to the pending one.
528        let batch = self.flush_and_reset();
529        self.apply_pending_schema();
530        batch
531    }
532
533    /// Returns the number of rows that can be added to this decoder before it is full.
534    pub fn capacity(&self) -> usize {
535        self.remaining_capacity
536    }
537
538    /// Returns true if the decoder has reached its capacity for the current batch.
539    pub fn batch_is_full(&self) -> bool {
540        self.remaining_capacity == 0
541    }
542
543    /// Returns true if the decoder has not decoded any batches yet (i.e., the current batch is empty).
544    pub fn batch_is_empty(&self) -> bool {
545        self.remaining_capacity == self.batch_size
546    }
547
548    // Decode either the block count or remaining capacity from `data` (an OCF block payload).
549    //
550    // Returns the number of bytes consumed from `data` along with the number of records decoded.
551    fn decode_block(&mut self, data: &[u8], count: usize) -> Result<(usize, usize), ArrowError> {
552        // OCF decoding never interleaves records across blocks, so no chunking.
553        let to_decode = std::cmp::min(count, self.remaining_capacity);
554        if to_decode == 0 {
555            return Ok((0, 0));
556        }
557        let consumed = self.active_decoder.decode(data, to_decode)?;
558        self.remaining_capacity -= to_decode;
559        Ok((consumed, to_decode))
560    }
561
562    // Produce a `RecordBatch` if at least one row is fully decoded, returning
563    // `Ok(None)` if no new rows are available.
564    fn flush_block(&mut self) -> Result<Option<RecordBatch>, ArrowError> {
565        self.flush_and_reset()
566    }
567}

569/// A builder that configures and constructs Avro readers and decoders.
570///
571/// `ReaderBuilder` is the primary entry point for this module. It supports:
572///
573/// * OCF reading via `Self::build`, returning a `Reader` over any `BufRead`;
574/// * streaming decoding via `Self::build_decoder`, returning a `Decoder`.
575///
576/// ### Options
577///
578/// * **`batch_size`**: Max rows per `RecordBatch` (default: `1024`). See `Self::with_batch_size`.
579/// * **`utf8_view`**: Use Arrow `StringViewArray` for string columns (default: `false`).
580///   See `Self::with_utf8_view`.
581/// * **`strict_mode`**: Opt‑in to stricter union handling (default: `false`).
582///   See `Self::with_strict_mode`.
583/// * **`reader_schema`**: Optional reader schema (projection / evolution) used when decoding
584///   values (default: `None`). See `Self::with_reader_schema`.
585/// * **`writer_schema_store`**: Required for building a `Decoder` for single‑object or
586///   Confluent framing. Maps fingerprints to Avro schemas. See `Self::with_writer_schema_store`.
587/// * **`active_fingerprint`**: Optional starting fingerprint for streaming decode when the
588///   first frame omits one (rare). See `Self::with_active_fingerprint`.
589///
590/// ### Examples
591///
592/// Read an OCF file in batches of 4096 rows:
593///
594/// ```no_run
595/// use std::fs::File;
596/// use std::io::BufReader;
597/// use arrow_avro::reader::ReaderBuilder;
598///
599/// let file = File::open("data.avro")?;
600/// let mut reader = ReaderBuilder::new()
601///     .with_batch_size(4096)
602///     .build(BufReader::new(file))?;
603/// # Ok::<(), Box<dyn std::error::Error>>(())
604/// ```

607///
608/// ```no_run
609/// use arrow_avro::schema::{AvroSchema, SchemaStore, Fingerprint, FingerprintAlgorithm};
610/// use arrow_avro::reader::ReaderBuilder;
611///

616///     .with_writer_schema_store(store)
617///     .build_decoder()?;
618/// # Ok::<(), Box<dyn std::error::Error>>(())
619/// ```

702    fn make_decoder(
703        &self,
704        header: Option<&Header>,
705        reader_schema: Option<&AvroSchema>,
706    ) -> Result<Decoder, ArrowError> {
707        if let Some(hdr) = header {
708            let writer_schema = hdr
709                .schema()
710                .map_err(|e| ArrowError::ExternalError(Box::new(e)))?
711                .ok_or_else(|| {
712                    ArrowError::ParseError("No Avro schema present in file header".into())
713                })?;
714            let record_decoder =
715                self.make_record_decoder_from_schemas(&writer_schema, reader_schema)?;
716            return Ok(self.make_decoder_with_parts(
717                record_decoder,
718                None,
719                IndexMap::new(),
720                FingerprintAlgorithm::Rabin,
721            ));
722        }
723        let store = self.writer_schema_store.as_ref().ok_or_else(|| {
724            ArrowError::ParseError("Writer schema store required for raw Avro".into())
725        })?;
726        let fingerprints = store.fingerprints();
727        if fingerprints.is_empty() {
728            return Err(ArrowError::ParseError(
729                "Writer schema store must contain at least one schema".into(),
730            ));
731        }
732        let start_fingerprint = self
733            .active_fingerprint
734            .or_else(|| fingerprints.first().copied())
735            .ok_or_else(|| {
736                ArrowError::ParseError("Could not determine initial schema fingerprint".into())
737            })?;
738        let mut cache = IndexMap::with_capacity(fingerprints.len().saturating_sub(1));
739        let mut active_decoder: Option<RecordDecoder> = None;
740        for fingerprint in store.fingerprints() {
741            let avro_schema = match store.lookup(&fingerprint) {
742                Some(schema) => schema,
743                None => {
744                    return Err(ArrowError::ComputeError(format!(
745                        "Fingerprint {fingerprint:?} not found in schema store",
746                    )));
747                }
748            };
749            let writer_schema = avro_schema.schema()?;
750            let record_decoder =
751                self.make_record_decoder_from_schemas(&writer_schema, reader_schema)?;
752            if fingerprint == start_fingerprint {
753                active_decoder = Some(record_decoder);
754            } else {
755                cache.insert(fingerprint, record_decoder);
756            }
757        }
758        let active_decoder = active_decoder.ok_or_else(|| {
759            ArrowError::ComputeError(format!(
760                "Initial fingerprint {start_fingerprint:?} not found in schema store"
761            ))
762        })?;
763        Ok(self.make_decoder_with_parts(
764            active_decoder,
765            Some(start_fingerprint),
766            cache,
767            store.fingerprint_algorithm(),
768        ))
769    }

875/// A high‑level Avro **Object Container File** reader.
876///
877/// `Reader` pulls blocks from a `BufRead` source, handles optional block compression,
878/// and decodes them row‑by‑row into Arrow `RecordBatch` values using an internal
879/// `Decoder`. It implements both:
880///
881/// * [`Iterator<Item = Result<RecordBatch, ArrowError>>`], and
882/// * `RecordBatchReader`, guaranteeing a consistent schema across all produced batches.
883///
884#[derive(Debug)]
885pub struct Reader<R: BufRead> {
886    reader: R,
887    header: Header,
888    decoder: Decoder,
889    block_decoder: BlockDecoder,
890    block_data: Vec<u8>,
891    block_count: usize,
892    block_cursor: usize,
893    finished: bool,
894}
895
896impl<R: BufRead> Reader<R> {
897    /// Returns the Arrow schema discovered from the Avro file header (or derived via
898    /// the optional reader schema).
899    pub fn schema(&self) -> SchemaRef {
900        self.decoder.schema()
901    }
902
903    /// Returns a reference to the parsed Avro container‑file header (magic, metadata, codec, sync).
904    pub fn avro_header(&self) -> &Header {
905        &self.header
906    }
907
908    /// Reads the next `RecordBatch` from the Avro file, or `Ok(None)` on EOF.
909    ///
910    /// Batches are bounded by `batch_size`; a single OCF block may yield multiple batches,
911    /// and a batch may also span multiple blocks.
912    fn read(&mut self) -> Result<Option<RecordBatch>, ArrowError> {
913        'outer: while !self.finished && !self.decoder.batch_is_full() {
914            while self.block_cursor == self.block_data.len() {
915                let buf = self.reader.fill_buf()?;
916                if buf.is_empty() {
917                    self.finished = true;
918                    break 'outer;
919                }
920                // Try to decode another block from the buffered reader.
921                let consumed = self.block_decoder.decode(buf)?;
922                self.reader.consume(consumed);
923                if let Some(block) = self.block_decoder.flush() {
924                    // Successfully decoded a block.
925                    self.block_data = if let Some(ref codec) = self.header.compression()? {
926                        codec.decompress(&block.data)?
927                    } else {
928                        block.data
929                    };
930                    self.block_count = block.count;
931                    self.block_cursor = 0;
932                } else if consumed == 0 {
933                    // The block decoder made no progress on a non-empty buffer.
934                    return Err(ArrowError::ParseError(
935                        "Could not decode next Avro block from partial data".to_string(),
936                    ));
937                }
938            }
939            // Decode as many rows as will fit in the current batch
940            if self.block_cursor < self.block_data.len() {
941                let (consumed, records_decoded) = self
942                    .decoder
943                    .decode_block(&self.block_data[self.block_cursor..], self.block_count)?;
944                self.block_cursor += consumed;
945                self.block_count -= records_decoded;
946            }
947        }
948        self.decoder.flush_block()
949    }
950}

952impl<R: BufRead> Iterator for Reader<R> {
953    type Item = Result<RecordBatch, ArrowError>;
954
955    fn next(&mut self) -> Option<Self::Item> {
956        self.read().transpose()
957    }
958}

1124    fn encode_zigzag(value: i64) -> Vec<u8> {
1125        let mut n = ((value << 1) ^ (value >> 63)) as u64;
1126        let mut out = Vec::new();
1127        loop {
1128            if (n & !0x7F) == 0 {
1129                out.push(n as u8);
1130                break;
1131            } else {
1132                out.push(((n & 0x7F) | 0x80) as u8);
1133                n >>= 7;
1134            }
1135        }
1136        out
1137    }
