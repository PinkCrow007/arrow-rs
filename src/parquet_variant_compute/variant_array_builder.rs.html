29/// A builder for [`VariantArray`]
30///
31/// This builder is used to construct a `VariantArray` and allows APIs for
32/// adding metadata
33///
34/// This builder always creates a `VariantArray` using [`BinaryViewArray`] for both
35/// the metadata and value fields.
36///
37/// # TODO
38/// 1. Support shredding: <https://github.com/apache/arrow-rs/issues/7895>
39///
40/// ## Example:
41/// ```
42/// # use arrow::array::Array;
43/// # use parquet_variant::{Variant, VariantBuilder, VariantBuilderExt};
44/// # use parquet_variant_compute::VariantArrayBuilder;
45/// // Create a new VariantArrayBuilder with a capacity of 100 rows
46/// let mut builder = VariantArrayBuilder::new(100);
47/// // append variant values
48/// builder.append_variant(Variant::from(42));
49/// // append a null row (note not a Variant::Null)
50/// builder.append_null();
51/// // append an object to the builder using VariantBuilderExt methods directly
52/// builder.new_object()
53///   .with_field("foo", "bar")
54///   .finish();
55///
56/// // create the final VariantArray
57/// let variant_array = builder.build();
58/// assert_eq!(variant_array.len(), 3);
59/// // // Access the values
60/// // row 1 is not null and is an integer
61/// assert!(!variant_array.is_null(0));
62/// assert_eq!(variant_array.value(0), Variant::from(42i32));
63/// // row 1 is null
64/// assert!(variant_array.is_null(1));
65/// // row 2 is not null and is an object
66/// assert!(!variant_array.is_null(2));
67/// let value = variant_array.value(2);
68/// let obj = value.as_object().expect("expected object");
69/// assert_eq!(obj.get("foo"), Some(Variant::from("bar")));
70/// ```
71#[derive(Debug)]
72pub struct VariantArrayBuilder {
73    /// Nulls
74    nulls: NullBufferBuilder,
75    /// builder for all the metadata
76    metadata_builder: WritableMetadataBuilder,
77    /// ending offset for each serialized metadata dictionary in the buffer
78    metadata_offsets: Vec<usize>,
79    /// builder for values
80    value_builder: ValueBuilder,
81    /// ending offset for each serialized variant value in the buffer
82    value_offsets: Vec<usize>,
83    /// The fields of the final `StructArray`
84    ///
85    /// TODO: 1) Add extension type metadata
86    /// TODO: 2) Add support for shredding
87    fields: Fields,
88}
89
90impl VariantArrayBuilder {
91    pub fn new(row_capacity: usize) -> Self {
92        // The subfields are expected to be non-nullable according to the parquet variant spec.
93        let metadata_field = Field::new("metadata", DataType::BinaryView, false);
94        let value_field = Field::new("value", DataType::BinaryView, false);
95
96        Self {
97            nulls: NullBufferBuilder::new(row_capacity),
98            metadata_builder: WritableMetadataBuilder::default(),
99            metadata_offsets: Vec::with_capacity(row_capacity),
100            value_builder: ValueBuilder::new(),
101            value_offsets: Vec::with_capacity(row_capacity),
102            fields: Fields::from(vec![metadata_field, value_field]),
103        }
104    }
105
106    /// Build the final builder
107    pub fn build(self) -> VariantArray {
108        let Self {
109            mut nulls,
110            metadata_builder,
111            metadata_offsets,
112            value_builder,
113            value_offsets,
114            fields,
115        } = self;
116
117        let metadata_buffer = metadata_builder.into_inner();
118        let metadata_array = binary_view_array_from_buffers(metadata_buffer, metadata_offsets);
119
120        let value_buffer = value_builder.into_inner();
121        let value_array = binary_view_array_from_buffers(value_buffer, value_offsets);
122
123        // The build the final struct array
124        let inner = StructArray::new(
125            fields,
126            vec![
127                Arc::new(metadata_array) as ArrayRef,
128                Arc::new(value_array) as ArrayRef,
129            ],
130            nulls.finish(),
131        );
132        // TODO add arrow extension type metadata
133
134        VariantArray::try_new(Arc::new(inner)).expect("valid VariantArray by construction")
135    }
136
137    /// Appends a null row to the builder.
138    pub fn append_null(&mut self) {
139        self.nulls.append_null();
140        // The subfields are expected to be non-nullable according to the parquet variant spec.
141        self.metadata_offsets.push(self.metadata_builder.offset());
142        self.value_offsets.push(self.value_builder.offset());
143    }
144
145    /// Append the [`Variant`] to the builder as the next row
146    pub fn append_variant(&mut self, variant: Variant) {
147        ValueBuilder::append_variant(self.parent_state(), variant);
148    }
149
150    /// Creates a builder-specific parent state
151    fn parent_state(&mut self) -> ParentState<'_, ArrayBuilderState<'_>> {
152        let state = ArrayBuilderState {
153            metadata_offsets: &mut self.metadata_offsets,
154            value_offsets: &mut self.value_offsets,
155            nulls: &mut self.nulls,
156        };
157
158        ParentState::new(&mut self.value_builder, &mut self.metadata_builder, state)
159    }
160}

162/// Builder-specific state for array building that manages array-level offsets and nulls. See
163/// [`VariantBuilderExt`] for details.
164#[derive(Debug)]
165pub struct ArrayBuilderState<'a> {
166    metadata_offsets: &'a mut Vec<usize>,
167    value_offsets: &'a mut Vec<usize>,
168    nulls: &'a mut NullBufferBuilder,
169}
170
171// All changes are pending until finalized
172impl BuilderSpecificState for ArrayBuilderState<'_> {
173    fn finish(
174        &mut self,
175        metadata_builder: &mut dyn MetadataBuilder,
176        value_builder: &mut ValueBuilder,
177    ) {
178        self.metadata_offsets.push(metadata_builder.finish());
179        self.value_offsets.push(value_builder.offset());
180        self.nulls.append_non_null();
181    }
182}
183
184impl VariantBuilderExt for VariantArrayBuilder {
185    type State<'a>
186        = ArrayBuilderState<'a>
187    where
188        Self: 'a;
189
190    /// Appending NULL to a variant array produces an actual NULL value
191    fn append_null(&mut self) {
192        self.append_null();
193    }
194
195    fn append_value<'m, 'v>(&mut self, value: impl Into<Variant<'m, 'v>>) {
196        self.append_variant(value.into());
197    }
198
199    fn try_new_list(&mut self) -> Result<ListBuilder<'_, Self::State<'_>>, ArrowError> {
200        Ok(ListBuilder::new(self.parent_state(), false))
201    }
202
203    fn try_new_object(&mut self) -> Result<ObjectBuilder<'_, Self::State<'_>>, ArrowError> {
204        Ok(ObjectBuilder::new(self.parent_state(), false))
205    }
206}

208fn binary_view_array_from_buffers(buffer: Vec<u8>, offsets: Vec<usize>) -> BinaryViewArray {
209    // All offsets are less than or equal to the buffer length, so we can safely cast all offsets
210    // inside the loop below, as long as the buffer length fits in u32.
211    u32::try_from(buffer.len()).expect("buffer length should fit in u32");
212
213    let mut builder = BinaryViewBuilder::with_capacity(offsets.len());
214    let block = builder.append_block(buffer.into());
215    // TODO this can be much faster if it creates the views directly during append
216    let mut start = 0;
217    for end in offsets {
218        let end = end as u32; // Safe cast: validated max offset fits in u32 above
219        builder
220            .try_append_view(block, start, end - start)
221            .expect("Failed to append view");
222        start = end;
223    }
224    builder.finish()
225}

227#[cfg(test)]
228mod test {
229    use super::*;
230    use arrow::array::Array;
231
232    /// Test that both the metadata and value buffers are non nullable
233    #[test]
234    fn test_variant_array_builder_non_nullable() {
235        let mut builder = VariantArrayBuilder::new(10);
236        builder.append_null(); // should not panic
237        builder.append_variant(Variant::from(42i32));
238        let variant_array = builder.build();
239
240        assert_eq!(variant_array.len(), 2);
241        assert!(variant_array.is_null(0));
242        assert!(!variant_array.is_null(1));
243        assert_eq!(variant_array.value(1), Variant::from(42i32));
244
245        // the metadata and value fields of non shredded variants should not be null
246        assert!(variant_array.metadata_field().nulls().is_none());
247        assert!(variant_array.value_field().unwrap().nulls().is_none());
248        let DataType::Struct(fields) = variant_array.data_type() else {
249            panic!("Expected VariantArray to have Struct data type");
250        };
251        for field in fields {
252            assert!(
253                !field.is_nullable(),
254                "Field {} should be non-nullable",
255                field.name()
256            );
257        }
258    }
259
260    /// Test using appending variants to the array builder
261    #[test]
262    fn test_variant_array_builder() {
263        let mut builder = VariantArrayBuilder::new(10);
264        builder.append_null(); // should not panic
265        builder.append_variant(Variant::from(42i32));
266
267        // make an object in the next row
268        builder.new_object().with_field("foo", "bar").finish();
269
270        // append a new list
271        builder
272            .new_list()
273            .with_value(Variant::from(1i32))
274            .with_value(Variant::from(2i32))
275            .finish();
276        let variant_array = builder.build();
277
278        assert_eq!(variant_array.len(), 4);
279        assert!(variant_array.is_null(0));
280        assert!(!variant_array.is_null(1));
281        assert_eq!(variant_array.value(1), Variant::from(42i32));
282        assert!(!variant_array.is_null(2));
283        let variant = variant_array.value(2);
284        let variant = variant.as_object().expect("variant to be an object");
285        assert_eq!(variant.get("foo").unwrap(), Variant::from("bar"));
286        assert!(!variant_array.is_null(3));
287        let variant = variant_array.value(3);
288        let list = variant.as_list().expect("variant to be a list");
289        assert_eq!(list.len(), 2);
290    }
