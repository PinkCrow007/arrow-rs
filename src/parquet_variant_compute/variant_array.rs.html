33/// An array of Parquet [`Variant`] values
34///
35/// A [`VariantArray`] wraps an Arrow [`StructArray`] that stores the underlying
36/// `metadata` and `value` fields, and adds convenience methods to access
37/// the `Variant`s
38///
39/// See [`VariantArrayBuilder`] for constructing a `VariantArray`.
40///
41/// [`VariantArrayBuilder`]: crate::VariantArrayBuilder
42///
43/// # Specification
44///
45/// 1. This code follows the conventions for storing variants in Arrow `StructArray`
46///    defined by [Extension Type for Parquet Variant arrow] and this [document].
47///    At the time of this writing, this is not yet a standardized Arrow extension type.
48///
49/// [Extension Type for Parquet Variant arrow]: https://github.com/apache/arrow/issues/46908
50/// [document]: https://docs.google.com/document/d/1pw0AWoMQY3SjD7R4LgbPvMjG_xSCtXp3rZHkVp9jpZ4/edit?usp=sharing
51#[derive(Debug)]
52pub struct VariantArray {
53    /// Reference to the underlying StructArray
54    inner: StructArray,
55
56    /// The metadata column of this variant
57    metadata: BinaryViewArray,
58
59    /// how is this variant array shredded?
60    shredding_state: ShreddingState,
61}
62
63impl VariantArray {
64    /// Creates a new `VariantArray` from a [`StructArray`].
65    ///
66    /// # Arguments
67    /// - `inner` - The underlying [`StructArray`] that contains the variant data.
68    ///
69    /// # Returns
70    /// - A new instance of `VariantArray`.
71    ///
72    /// # Errors:
73    /// - If the `StructArray` does not contain the required fields
74    ///
75    /// # Requirements of the `StructArray`
76    ///
77    /// 1. A required field named `metadata` which is binary, large_binary, or
78    ///    binary_view
79    ///
80    /// 2. An optional field named `value` that is binary, large_binary, or
81    ///    binary_view
82    ///
83    /// 3. An optional field named `typed_value` which can be any primitive type
84    ///    or be a list, large_list, list_view or struct
85    ///
86    /// NOTE: It is also permissible for the metadata field to be
87    /// Dictionary-Encoded, preferably (but not required) with an index type of
88    /// int8.
89    ///
90    /// Currently, only [`BinaryViewArray`] are supported.
91    pub fn try_new(inner: ArrayRef) -> Result<Self, ArrowError> {
92        let Some(inner) = inner.as_struct_opt() else {
93            return Err(ArrowError::InvalidArgumentError(
94                "Invalid VariantArray: requires StructArray as input".to_string(),
95            ));
96        };
97
98        // Note the specification allows for any order so we must search by name
99
100        // Ensure the StructArray has a metadata field of BinaryView
101        let Some(metadata_field) = inner.column_by_name("metadata") else {
102            return Err(ArrowError::InvalidArgumentError(
103                "Invalid VariantArray: StructArray must contain a 'metadata' field".to_string(),
104            ));
105        };
106        let Some(metadata) = metadata_field.as_binary_view_opt() else {
107            return Err(ArrowError::NotYetImplemented(format!(
108                "VariantArray 'metadata' field must be BinaryView, got {}",
109                metadata_field.data_type()
110            )));
111        };
112
113        // Extract value and typed_value fields
114        let value = if let Some(value_col) = inner.column_by_name("value") {
115            if let Some(binary_view) = value_col.as_binary_view_opt() {
116                Some(binary_view.clone())
117            } else {
118                return Err(ArrowError::NotYetImplemented(format!(
119                    "VariantArray 'value' field must be BinaryView, got {}",
120                    value_col.data_type()
121                )));
122            }
123        } else {
124            None
125        };
126        let typed_value = inner.column_by_name("typed_value").cloned();
127
128        // Note these clones are cheap, they just bump the ref count
129        Ok(Self {
130            inner: inner.clone(),
131            metadata: metadata.clone(),
132            shredding_state: ShreddingState::try_new(value, typed_value)?,
133        })
134    }
135
136    pub(crate) fn from_parts(
137        metadata: BinaryViewArray,
138        value: Option<BinaryViewArray>,
139        typed_value: Option<ArrayRef>,
140        nulls: Option<NullBuffer>,
141    ) -> Self {
142        let mut builder =
143            StructArrayBuilder::new().with_field("metadata", Arc::new(metadata.clone()), false);
144        if let Some(value) = value.clone() {
145            builder = builder.with_field("value", Arc::new(value), true);
146        }
147        if let Some(typed_value) = typed_value.clone() {
148            builder = builder.with_field("typed_value", typed_value, true);
149        }
150        if let Some(nulls) = nulls {
151            builder = builder.with_nulls(nulls);
152        }
153
154        // This would be a lot simpler if ShreddingState were just a pair of Option... we already
155        // have everything we need.
156        let inner = builder.build();
157        let shredding_state = ShreddingState::try_new(value, typed_value).unwrap(); // valid by construction
158        Self {
159            inner,
160            metadata,
161            shredding_state,
162        }
163    }

180    /// Return the [`Variant`] instance stored at the given row
181    ///
182    /// Note: This method does not check for nulls and the value is arbitrary
183    /// (but still well-defined) if [`is_null`](Self::is_null) returns true for the index.
184    ///
185    /// # Panics
186    /// * if the index is out of bounds
187    /// * if the array value is null
188    ///
189    /// If this is a shredded variant but has no value at the shredded location, it
190    /// will return [`Variant::Null`].
191    ///
192    ///
193    /// # Performance Note
194    ///
195    /// This is certainly not the most efficient way to access values in a
196    /// `VariantArray`, but it is useful for testing and debugging.
197    ///
198    /// Note: Does not do deep validation of the [`Variant`], so it is up to the
199    /// caller to ensure that the metadata and value were constructed correctly.
200    pub fn value(&self, index: usize) -> Variant<'_, '_> {
201        match &self.shredding_state {
202            ShreddingState::Unshredded { value, .. } => {
203                // Unshredded case
204                Variant::new(self.metadata.value(index), value.value(index))
205            }
206            ShreddingState::Typed { typed_value, .. } => {
207                // Typed case (formerly PerfectlyShredded)
208                if typed_value.is_null(index) {
209                    Variant::Null
210                } else {
211                    typed_value_to_variant(typed_value, index)
212                }
213            }
214            ShreddingState::PartiallyShredded {
215                value, typed_value, ..
216            } => {
217                // PartiallyShredded case (formerly ImperfectlyShredded)
218                if typed_value.is_null(index) {
219                    Variant::new(self.metadata.value(index), value.value(index))
220                } else {
221                    typed_value_to_variant(typed_value, index)
222                }
223            }
224            ShreddingState::AllNull => {
225                // AllNull case: neither value nor typed_value fields exist
226                // NOTE: This handles the case where neither value nor typed_value fields exist.
227                // For top-level variants, this returns Variant::Null (JSON null).
228                // For shredded object fields, this technically should indicate SQL NULL,
229                // but the current API cannot distinguish these contexts.
230                Variant::Null
231            }
232        }
233    }

247        self.shredding_state.typed_value_field()

251/// One shredded field of a partially or prefectly shredded variant. For example, suppose the
252/// shredding schema for variant `v` treats it as an object with a single field `a`, where `a` is
253/// itself a struct with the single field `b` of type INT. Then the physical layout of the column
254/// is:
255///
256/// ```text
257/// v: VARIANT {
258///     metadata: BINARY,
259///     value: BINARY,
260///     typed_value: STRUCT {
261///         a: SHREDDED_VARIANT_FIELD {
262///             value: BINARY,
263///             typed_value: STRUCT {
264///                 a: SHREDDED_VARIANT_FIELD {
265///                     value: BINARY,
266///                     typed_value: INT,
267///                 },
268///             },
269///         },
270///     },
271/// }
272/// ```
273///
274/// In the above, each row of `v.value` is either a variant value (shredding failed, `v` was not an
275/// object at all) or a variant object (partial shredding, `v` was an object but included unexpected
276/// fields other than `a`), or is NULL (perfect shredding, `v` was an object containing only the
277/// single expected field `a`).
278///
279/// A similar story unfolds for each `v.typed_value.a.value` -- a variant value if shredding failed
280/// (`v:a` was not an object at all), or a variant object (`v:a` was an object with unexpected
281/// additional fields), or NULL (`v:a` was an object containing only the single expected field `b`).
282///
283/// Finally, `v.typed_value.a.typed_value.b.value` is either NULL (`v:a.b` was an integer) or else a
284/// variant value (which could be `Variant::Null`).
285#[derive(Debug)]
286pub struct ShreddedVariantFieldArray {
287    /// Reference to the underlying StructArray
288    inner: StructArray,
289    shredding_state: ShreddingState,
290}
291
292#[allow(unused)]
293impl ShreddedVariantFieldArray {
294    /// Creates a new `ShreddedVariantFieldArray` from a [`StructArray`].
295    ///
296    /// # Arguments
297    /// - `inner` - The underlying [`StructArray`] that contains the variant data.
298    ///
299    /// # Returns
300    /// - A new instance of `ShreddedVariantFieldArray`.
301    ///
302    /// # Errors:
303    /// - If the `StructArray` does not contain the required fields
304    ///
305    /// # Requirements of the `StructArray`
306    ///
307    /// 1. An optional field named `value` that is binary, large_binary, or
308    ///    binary_view
309    ///
310    /// 2. An optional field named `typed_value` which can be any primitive type
311    ///    or be a list, large_list, list_view or struct
312    ///
313    /// Currently, only `value` columns of type [`BinaryViewArray`] are supported.
314    pub fn try_new(inner: ArrayRef) -> Result<Self, ArrowError> {
315        let Some(inner_struct) = inner.as_struct_opt() else {
316            return Err(ArrowError::InvalidArgumentError(
317                "Invalid ShreddedVariantFieldArray: requires StructArray as input".to_string(),
318            ));
319        };
320
321        // Extract value and typed_value fields (metadata is not expected in ShreddedVariantFieldArray)
322        let value = inner_struct
323            .column_by_name("value")
324            .and_then(|col| col.as_binary_view_opt().cloned());
325        let typed_value = inner_struct.column_by_name("typed_value").cloned();
326
327        // Note this clone is cheap, it just bumps the ref count
328        let inner = inner_struct.clone();
329        Ok(Self {
330            inner: inner.clone(),
331            shredding_state: ShreddingState::try_new(value, typed_value)?,
332        })
333    }
334
335    /// Return the shredding state of this `VariantArray`
336    pub fn shredding_state(&self) -> &ShreddingState {
337        &self.shredding_state
338    }
339
340    /// Return a reference to the value field of the `StructArray`
341    pub fn value_field(&self) -> Option<&BinaryViewArray> {
342        self.shredding_state.value_field()
343    }
344
345    /// Return a reference to the typed_value field of the `StructArray`, if present
346    pub fn typed_value_field(&self) -> Option<&ArrayRef> {
347        self.shredding_state.typed_value_field()
348    }
349
350    /// Returns a reference to the underlying [`StructArray`].
351    pub fn inner(&self) -> &StructArray {
352        &self.inner
353    }
354}

410/// Represents the shredding state of a [`VariantArray`]
411///
412/// [`VariantArray`]s can be shredded according to the [Parquet Variant
413/// Shredding Spec]. Shredding means that the actual value is stored in a typed
414/// `typed_field` instead of the generic `value` field.
415///
416/// Both value and typed_value are optional fields used together to encode a
417/// single value. Values in the two fields must be interpreted according to the
418/// following table (see [Parquet Variant Shredding Spec] for more details):
419///
420/// | value | typed_value | Meaning |
421/// |----------|--------------|---------|
422/// | null     | null         | The value is missing; only valid for shredded object fields |
423/// | non-null | null         | The value is present and may be any type, including `null` |
424/// | null     | non-null     | The value is present and is the shredded type |
425/// | non-null | non-null     | The value is present and is a partially shredded object |
426///
427/// [Parquet Variant Shredding Spec]: https://github.com/apache/parquet-format/blob/master/VariantShredding.md#value-shredding
428#[derive(Debug)]
429pub enum ShreddingState {
430    /// This variant has no typed_value field
431    Unshredded { value: BinaryViewArray },
432    /// This variant has a typed_value field and no value field
433    /// meaning it is the shredded type
434    Typed { typed_value: ArrayRef },
435    /// Imperfectly shredded: Shredded values reside in `typed_value` while those that failed to
436    /// shred reside in `value`. Missing field values are NULL in both columns, while NULL primitive
437    /// values have NULL `typed_value` and `Variant::Null` in `value`.
438    ///
439    /// NOTE: A partially shredded struct is a special kind of imperfect shredding, where
440    /// `typed_value` and `value` are both non-NULL. The `typed_value` is a struct containing the
441    /// subset of fields for which shredding was attempted (each field will then have its own value
442    /// and/or typed_value sub-fields that indicate how shredding actually turned out). Meanwhile,
443    /// the `value` is a variant object containing the subset of fields for which shredding was
444    /// not even attempted.
445    PartiallyShredded {
446        value: BinaryViewArray,
447        typed_value: ArrayRef,
448    },
449    /// All values are null, only metadata is present.
450    ///
451    /// This state occurs when neither `value` nor `typed_value` fields exist in the schema.
452    /// Note: By strict spec interpretation, this should only be valid for shredded object fields,
453    /// not top-level variants. However, we allow it and treat as Variant::Null for pragmatic
454    /// handling of missing data.
455    AllNull,
456}
457
458impl ShreddingState {
459    /// try to create a new `ShreddingState` from the given fields
460    pub fn try_new(
461        value: Option<BinaryViewArray>,
462        typed_value: Option<ArrayRef>,
463    ) -> Result<Self, ArrowError> {
464        match (value, typed_value) {
465            (Some(value), Some(typed_value)) => Ok(Self::PartiallyShredded { value, typed_value }),
466            (Some(value), None) => Ok(Self::Unshredded { value }),
467            (None, Some(typed_value)) => Ok(Self::Typed { typed_value }),
468            (None, None) => Ok(Self::AllNull),
469        }
470    }
471
472    /// Return a reference to the value field, if present
473    pub fn value_field(&self) -> Option<&BinaryViewArray> {
474        match self {
475            ShreddingState::Unshredded { value, .. } => Some(value),
476            ShreddingState::Typed { .. } => None,
477            ShreddingState::PartiallyShredded { value, .. } => Some(value),
478            ShreddingState::AllNull => None,
479        }
480    }
481
482    /// Return a reference to the typed_value field, if present
483    pub fn typed_value_field(&self) -> Option<&ArrayRef> {
484        match self {
485            ShreddingState::Unshredded { .. } => None,
486            ShreddingState::Typed { typed_value, .. } => Some(typed_value),
487            ShreddingState::PartiallyShredded { typed_value, .. } => Some(typed_value),
488            ShreddingState::AllNull => None,
489        }
490    }
491
492    /// Slice all the underlying arrays
493    pub fn slice(&self, offset: usize, length: usize) -> Self {
494        match self {
495            ShreddingState::Unshredded { value } => ShreddingState::Unshredded {
496                value: value.slice(offset, length),
497            },
498            ShreddingState::Typed { typed_value } => ShreddingState::Typed {
499                typed_value: typed_value.slice(offset, length),
500            },
501            ShreddingState::PartiallyShredded { value, typed_value } => {
502                ShreddingState::PartiallyShredded {
503                    value: value.slice(offset, length),
504                    typed_value: typed_value.slice(offset, length),
505                }
506            }
507            ShreddingState::AllNull => ShreddingState::AllNull,
508        }
509    }
510}

512/// Builds struct arrays from component fields
513///
514/// TODO: move to arrow crate
515#[derive(Debug, Default, Clone)]
516pub(crate) struct StructArrayBuilder {
517    fields: Vec<FieldRef>,
518    arrays: Vec<ArrayRef>,
519    nulls: Option<NullBuffer>,
520}
521
522impl StructArrayBuilder {
523    pub fn new() -> Self {
524        Default::default()
525    }
526
527    /// Add an array to this struct array as a field with the specified name.
528    pub fn with_field(mut self, field_name: &str, array: ArrayRef, nullable: bool) -> Self {
529        let field = Field::new(field_name, array.data_type().clone(), nullable);
530        self.fields.push(Arc::new(field));
531        self.arrays.push(array);
532        self
533    }
534
535    /// Set the null buffer for this struct array.
536    pub fn with_nulls(mut self, nulls: NullBuffer) -> Self {
537        self.nulls = Some(nulls);
538        self
539    }
540
541    pub fn build(self) -> StructArray {
542        let Self {
543            fields,
544            arrays,
545            nulls,
546        } = self;
547        StructArray::new(Fields::from(fields), arrays, nulls)
548    }
549}

551/// returns the non-null element at index as a Variant
552fn typed_value_to_variant(typed_value: &ArrayRef, index: usize) -> Variant<'_, '_> {
553    match typed_value.data_type() {
554        DataType::Boolean => {
555            let boolean_array = typed_value.as_boolean();
556            let value = boolean_array.value(index);
557            Variant::from(value)
558        }
559        DataType::FixedSizeBinary(binary_len) => {
560            let array = typed_value.as_fixed_size_binary();
561            // Try to treat 16 byte FixedSizeBinary as UUID
562            let value = array.value(index);
563            if *binary_len == 16 {
564                if let Ok(uuid) = Uuid::from_slice(value) {
565                    return Variant::from(uuid);
566                }
567            }
568            let value = array.value(index);
569            Variant::from(value)
570        }
571        DataType::BinaryView => {
572            let array = typed_value.as_binary_view();
573            let value = array.value(index);
574            Variant::from(value)
575        }
576        DataType::Utf8 => {
577            let array = typed_value.as_string::<i32>();
578            let value = array.value(index);
579            Variant::from(value)
580        }
581        DataType::Int8 => {
582            primitive_conversion_single_value!(Int8Type, typed_value, index)
583        }
584        DataType::Int16 => {
585            primitive_conversion_single_value!(Int16Type, typed_value, index)
586        }
587        DataType::Int32 => {
588            primitive_conversion_single_value!(Int32Type, typed_value, index)
589        }
590        DataType::Int64 => {
591            primitive_conversion_single_value!(Int64Type, typed_value, index)
592        }
593        DataType::UInt8 => {
594            primitive_conversion_single_value!(UInt8Type, typed_value, index)
595        }
596        DataType::UInt16 => {
597            primitive_conversion_single_value!(UInt16Type, typed_value, index)
598        }
599        DataType::UInt32 => {
600            primitive_conversion_single_value!(UInt32Type, typed_value, index)
601        }
602        DataType::UInt64 => {
603            primitive_conversion_single_value!(UInt64Type, typed_value, index)
604        }
605        DataType::Float16 => {
606            primitive_conversion_single_value!(Float16Type, typed_value, index)
607        }
608        DataType::Float32 => {
609            primitive_conversion_single_value!(Float32Type, typed_value, index)
610        }
611        DataType::Float64 => {
612            primitive_conversion_single_value!(Float64Type, typed_value, index)
613        }
614        // todo other types here (note this is very similar to cast_to_variant.rs)
615        // so it would be great to figure out how to share this code
616        _ => {
617            // We shouldn't panic in production code, but this is a
618            // placeholder until we implement more types
619            // https://github.com/apache/arrow-rs/issues/8091
620            debug_assert!(
621                false,
622                "Unsupported typed_value type: {:?}",
623                typed_value.data_type()
624            );
625            Variant::Null
626        }
627    }
628}
